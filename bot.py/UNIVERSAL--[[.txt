    c.Parent = btn

    s.Color = Color3.fromRGB(255, 0, 255)

    s.Parent = btn

    btn.MouseEnter:Connect(function()
        TweenService:Create(btn, TweenInfo.new(0.2), {BackgroundColor3 = Color3.fromRGB(120, 0, 120)}):Play()
    end)
    btn.MouseLeave:Connect(function()
        TweenService:Create(btn, TweenInfo.new(0.2), {BackgroundColor3 = Color3.fromRGB(40, 40, 60)}):Play()
    end)

    btn.MouseButton1Click:Connect(function()
        pcall(callback)
        notify(name .. " EXECUTED", Color3.fromRGB(0, 255, 0))
    end)
end

    task.wait(2.2)

-- === ANIM PLAY FUNCTION ===
local function playAnim(id)
    pcall(function()
        if Animator then
            local anim = Instance.new("Animation")
            anim.AnimationId = "rbxassetid://"..id
            local track = Animator:LoadAnimation(anim)
            track:Play()
        end
    end)
end
